Program 10: //Create a package in project by name "mypack". //Type the following program with filename: Circle.java.

Description: This program demonstrates a basic implementation of an array traversal.

Code:

           
                package mypack;

                public class Circle {

                    int r;
                    
                    public Circle () {
                        this.r = 5;
                    }
                    
                    public Circle(int r){
                        this.r = r;
                    }
                    
                    public double area() {
                        return 3.14 * r * r;
                    }
                }

                //Create a package in project by name "demo".
                //Type the following program with filename: Demo.java

                package demo;

                import mypack.Circle;

                public class Demo {
                    public static void main(String args[]) {
                        
                        Circle c = new Circle();
                    
                        System.out.println("Area of Circle: "+ c.area());
                    }	
                }
