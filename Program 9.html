9. Develop a JAVA program to raise a custom exception (user defined exception) for DivisionByZero using try, catch, throw and finally.

Description: This program demonstrates a basic implementation of an array traversal.

Code:


import java.util.Scanner;

class CustomException extends Exception {

	public String toString() {
		return "CustomException: Denominator is Zero";
	}
} //end CustomExpection

public class CustomExceptionDemo {
	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		
		int a, b, c;
		
		System.out.printf("Enter a: ");
		a = scan.nextInt();
		
		System.out.printf("Enter b: ");
		b = scan.nextInt();
		
		try {
			if(b == 0)
				throw new CustomException();
			
			c = a / b;
			System.out.println("c = "+c);
		}
		catch (CustomException e) {
			b = 1;
			c = a / b;
			System.out.println("DivideByZero Exception caught and handled:  ");
			System.out.println("c = " + c);
			//System.out.printf("%s",e);
		}
		finally {
			System.out.println("Finally block.");
		}
	} //end main
} //end ExceptionDemo
